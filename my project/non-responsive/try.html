const promise = new Promise((resolve, reject) => { // Asynchronous operation setTimeout(() => { const success = true; // Simulate success or failure if (success) { resolve("Operation was successful!"); } else { reject("Operation failed!"); } }, 2000); }); promise .then((message) => { console.log(message); // Output: "Operation was successful!" }) .catch((error) => { console.error(error); // Output: "Operation failed!" }) .finally(() => { console.log("Operation completed."); // Always executed });
